---
title: 'Pro Stream'
api: 'GET /stream/pro'
description: 'Cleaned and flattened swap data optimized for trading bots'
---

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token: `Bearer YOUR_API_KEY`
</ParamField>

## Response

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp of the swap event
</ResponseField>

<ResponseField name="token" type="object">
  Token information
  <Expandable title="properties">
    <ResponseField name="address" type="string">
      Token contract address
    </ResponseField>
    <ResponseField name="symbol" type="string">
      Token symbol (e.g., "USDC")
    </ResponseField>
    <ResponseField name="name" type="string">
      Full token name
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pair" type="object">
  Trading pair information
  <Expandable title="properties">
    <ResponseField name="address" type="string">
      Pair contract address
    </ResponseField>
    <ResponseField name="dex" type="string">
      DEX name (Raydium, Orca, etc.)
    </ResponseField>
    <ResponseField name="price_usd" type="number">
      Current price in USD
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="window_metrics" type="object">
  Trading metrics by time window (m5, h1, h6, h24)
</ResponseField>
